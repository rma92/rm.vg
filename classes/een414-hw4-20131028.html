Experiment 4: MESSAGE DISPLAY

Abstract

The object of this experiment was to produce a 7-segment message display, that accepts a phone number input. It is to support scrolling and blinking. It is implemented as a shift register. Our implementation gets the effect of a shift register by shifting the display, while maintaining a buffer in memory.

A number may be entered using the 4 right-most switches, and pressing the center button. The display may be scrolled left and right by pressing the left and right buttons. Pressing the up button resets everything. Enabling the second to left-most switch causes the display to blink.

module sev_seg_with_clk_top(
    input [7:0] num_in,
    input [3:0] cmdl,
         input enter_num_button,
    input clk_main,
         input reset,
    output [7:0] sev_seg_leds,
    output [7:0] led_enable
    );

         wire clk_slw ;

         reg enable_blink;
         reg blink_state;
         reg movel; //this register, if 1, is to scroll automatically. if 0, nothing else.

         reg [14:0]clk_div ;

         reg[31:0] counter;

         reg [7:0] iram[0:`memsize];

         reg [7:0] num_1;
         reg [7:0] num_2;
         reg [7:0] num_3;
         reg [7:0] num_4;
         reg [7:0] num_5;
         reg [7:0] num_6;
         reg [7:0] num_7;
         reg [7:0] num_8;

         reg [3:0] start_mem; //the initial index to read from.
         reg [3:0] write_loc; //the initial index to write from. Incremented after.

         integer i;

         sev_seg_with_clk mod_1(
      .num_1(num_1),                                                            // input number one
      .num_2(num_2),                                                            // input number two
                .num_3(num_3),                                                                // input number two
                .num_4(num_4),                                                                // input number two
                .num_5(num_5),                                                                // input number two
                .num_6(num_6),                                                                // input number two
                .num_7(num_7),                                                                // input number two
                .num_8(num_8),                                                                // input number two
                .clk(clk_div[14]),                                            // we need a clk to switch between multiple digits
                .reset(reset),
           .sev_seg_leds(sev_seg_leds),                        // controls the eight individual LEDs in all sev_seg digits.
           .led_enable(led_enable)
                );

                //---------------- clock for seven seg control------------------

                ip_clk_div instance_name(                             //generated core to obtain a slower clk of 5MHz from onboard 100 MHz oscillator.
                .CLK_IN1(clk_main),
                 .CLK_OUT1(clk_slw));

         initial
                begin
                        num_1 <= " ";
                        num_2 <= " ";
                        num_3 <= " ";
                        num_4 <= " ";
                        num_5 <= " ";
                        num_6 <= " ";
                        num_7 <= " ";
                        num_8 <= " ";
                        start_mem <= 0;
                        for( i=0; i < `memsize - 1; i = i + 1 ) begin
                                        iram[i] = " ";
                        end
                        iram[0] = "E";
                        iram[1] = "N";
                        iram[2] = "T";
                        iram[3] = "E";
                        iram[4] = "R";
                        iram[5] = " ";
                        iram[6] = "A";
                        iram[7] = " ";
                        iram[8] = "N";
                        iram[9] = "U";
                        iram[10] = "M";
                        iram[11] = "B";
                        iram[12] = "E";
                        iram[13] = "R";
                        movel = 1;
                end



        always @ (posedge clk_slw)                     // clock divider by 2^15 
         begin
                if (reset) begin
                        clk_div <= 15'd0 ;

                        iram[0] = "E";
                        iram[1] = "N";
                        iram[2] = "T";
                        iram[3] = "E";
                        iram[4] = "R";
                        iram[5] = " ";
                        iram[6] = "A";
                        iram[7] = " ";
                        iram[8] = "N";
                        iram[9] = "U";
                        iram[10] = "M";
                        iram[11] = "B";
                        iram[12] = "E";
                        iram[13] = "R";
                        start_mem = 0;
                        write_loc = 0;
                        movel = 1;
                end
                else
                        clk_div <= clk_div + 15'd1 ;
                counter = counter + 1;

                if(counter > 31'h400000) begin //5 million = 4C 4B40
                        counter = 0;
                        blink_state = !blink_state;
                        //0 = do noth
                        if(movel == 1) begin
                                //move left (auto for initial message)
                                start_mem = start_mem + 1;
                                if(start_mem > 6)
                                        start_mem = 0;
                        end


                        if(cmdl == 4'h1) begin
                                //move left
                                start_mem = start_mem + 1;
                        end

                        else if(cmdl == 4'h2) begin
                                //move right
                                start_mem = start_mem - 1;
                        end

                        else if(cmdl == 4'h8) begin
                                //backspace
                                if(write_loc > 1) begin
                                        write_loc = write_loc - 1;
                                end
                        end
                        else if(enter_num_button == 1) begin
                                //insert digit
                                movel = 0;
                                if( write_loc == 0) begin
                                        for( i=0; i < `memsize - 1; i = i + 1 ) begin
                                                iram[i] = " ";
                                        end

                                        start_mem = 0;
                                end
                                if( write_loc == 3 ) begin
                                        iram[write_loc] = "-";
                                        write_loc = write_loc + 1;
                                end
                                else if(write_loc == 7) begin
                                        iram[write_loc] = "-";
                                        write_loc = write_loc + 1;
                                end

                                iram[write_loc] = num_in;
                                iram[write_loc + 1] = ".";
                                write_loc = write_loc + 1;
                        end
                end

                enable_blink <= (cmdl == 4'h4)?1:0;

                if(enable_blink == 1 && blink_state == 1) begin
                        num_8 <= " ";
                        num_7 <= " ";
                        num_6 <= " ";
                        num_5 <= " ";
                        num_4 <= " ";
                        num_3 <= " ";
                        num_2 <= " ";
                        num_1 <= " ";
                end
                else begin
                        num_8 <= iram[start_mem];
                        num_7 <= iram[start_mem+1];
                        num_6 <= iram[start_mem+2];
                        num_5 <= iram[start_mem+3];
                        num_4 <= iram[start_mem+4];
                        num_3 <= iram[start_mem+5];
                        num_2 <= iram[start_mem+6];
                        num_1 <= iram[start_mem+7];
                end
         end
endmodule
